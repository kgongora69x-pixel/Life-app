<script>

const dias = ["domingo","lunes","martes","miercoles","jueves","viernes","sabado"];
const hoyIndex = new Date().getDay();
const hoy = dias[hoyIndex];

document.getElementById("fecha").innerText = hoy.toUpperCase();

function cursosDia() {
  const mapa = {
    lunes: ["Botánica","Química"],
    martes: ["Física","Matemática"],
    miercoles: ["Botánica","Química"],
    jueves: ["Física","Matemática"],
    viernes: ["Botánica","Química"],
    sabado: ["Física","Matemática"]
  };
  return mapa[hoy] || [];
}

function rutinaDia() {

let base = [
["06:00","Barrer"],
["06:30","Tareas"],
["06:45","Ducha"],
["07:00","Desayuno"],
["07:15","Traslado"]
];

if (["lunes","jueves","sabado"].includes(hoy)) {
  base.push(["14:00","Deporte"]);
} else {
  base.push(["14:00","Trabajo"]);
}

const cursos = cursosDia();

base.push(["20:00","Curso fuerte: "+cursos[0]]);
base.push(["21:15","Descanso"]);
base.push(["21:20","Curso ligero: "+cursos[1]]);

return base;
}

function cargar() {
  const cont = document.getElementById("rutina");
  cont.innerHTML = "";

  const rutina = rutinaDia();

  let completadas = 0;

  rutina.forEach((item,i)=>{
    const key = hoy+"_"+i;
    const estado = localStorage.getItem(key);

    if (estado) completadas++;

    cont.innerHTML += `
      <div class="card ${estado ? 'hecho':''}">
        <div class="hora">${item[0]}</div>
        <div>${item[1]}</div>
        <button onclick="ok('${key}')">✔</button>
      </div>
    `;
  });

  let porcentaje = (completadas / rutina.length) * 100;
  document.getElementById("barra").style.width = porcentaje + "%";
}

function ok(key){
  localStorage.setItem(key,true);
  cargar();
}

function ver(tab){
  document.getElementById("rutina").classList.add("hidden");
  document.getElementById("reflexion").classList.add("hidden");
  document.getElementById(tab).classList.remove("hidden");
}

function guardar(){
  localStorage.setItem("aprendi_"+hoy,document.getElementById("aprendi").value);
  localStorage.setItem("no_"+hoy,document.getElementById("no").value);
  alert("Guardado ✅");
}

cargar();
